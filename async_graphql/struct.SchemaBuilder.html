pub fn extension(self, extension: impl ExtensionFactory) -> Self

Add an extension to the schema.

Examples
use async_graphql::*;

struct Query;

#[Object]
impl Query {
    async fn value(&self) -> i32 {
        100
    }
}

let schema = Schema::build(Query, EmptyMutation, EmptySubscription)
    .extension(extensions::Logger)
    .finish();
